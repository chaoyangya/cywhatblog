# encoding=utf-8
import sys
def get_python_info():
    print('当前文件名 {} '.format(sys._getframe().f_code.co_filename))
    print('所属函数名 {} '.format(sys._getframe().f_code.co_name))
    print('第 {} 行 '.format(sys._getframe().f_lineno))
    
# 执行
get_python_info()
